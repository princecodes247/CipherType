How to use

Simple Typing

let instance = new CipherType("#target");

instance
.type("Simple text")

Typing multiple lines at a time

Method 1

let instance = new CipherType("#target");
instance .type(["Simple text", "Kept simple"])

Method 2

let instance = new CipherType("#target");
instance
.type("Simple text")
.type("Kept simple")

Deleting text

let instance = new CipherType("#target");
instance
.type("Changing text")
.delete()
.type("No...")
.delete()
.type("Delete test")
.delete(4)
.type("text!")

Using cursor

let instance = new CipherType("#target", {withCursor: true});
instance
.type("With cursor")

Typing text straight

let instance = new CipherType("#target", {switchCount: 1});

instance
.type("Straight to the point...", 100) .run()
